def self.included(group)
  group.class_eval do
    let :pairs do
      []
    end

    before :each do
      Mattock::CommandLine.should_receive(:execute) do |cmd|
        pattern, res = pairs.shift
        pattern =~ cmd
        Mattock::MockCommandResult.create(*res)
      end.any_number_of_times
    end

    after :each do
      pairs.should have_all_been_called
    end
  end
end

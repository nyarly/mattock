def inspect
  "<path: #{
    if field_unset?(:absolute_path)
      if field_unset?(:relative_path)
        "<<?>>"
      else
        "?/#{relative_path}"
      end
    else
      absolute_path.inspect
    end
  }>"
end

def pathname
  @pathname ||=
    begin
      fail_unless_set(:absolute_path)
      require 'pathname'
      Pathname.new(absolute_path)
    end
end

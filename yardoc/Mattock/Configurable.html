def check_required
  missing = self.class.missing_required_fields_on(self)
  unless missing.empty?
    raise "Required field#{missing.length > 1 ? "s" : ""} #{missing.map{|field| field.to_s.inspect}.join(", ")} unset on #{self.inspect}"
  end
  self
end
